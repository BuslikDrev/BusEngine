Настройка окружения

После просмотра урока, за основу возьмите плагин в папке "BusEngine/Code/Plugin" измените везде Plugin на своё название, пропишите путь к своему плагину в файле настройки проекта "*.busproject" или "busengine.busengine" и можете приступать к творению!

Пример файла Program.cs

В коде отражены все методы своего плагина с названием MyPlugin.

  1. /** API BusEngine.Game - пользовательский код */
  2. namespace BusEngine.Game {
  3.     /** API BusEngine.Plugin */
  4.     public class MyPlugin : BusEngine.Plugin {
  5.         // при запуске BusEngine до создания формы
  6.         public override void Initialize() {
  7.             BusEngine.Log.Info("MyPlugin Initialize");
  8.         }
  10.         // после загрузки определённого плагина
  11.         public override void Initialize(string plugin) {
  12.             BusEngine.Log.Info("MyPlugin Initialize: {0}", plugin);
  13.         }
  15.         // при запуске BusEngine после создания формы Canvas
  16.         public override void InitializeCanvas() {
  17.             BusEngine.Log.Info("MyPlugin InitializeCanvas");
  18.         }
  20.         // перед закрытием BusEngine
  21.         public override void Shutdown() {
  22.             BusEngine.Log.Info("MyPlugin Shutdown");
  23.         }
  25.         // перед загрузкой игрового уровня
  26.         public override void OnLevelLoading(string level) {
  27.             BusEngine.Log.Info("MyPlugin OnLevelLoading: {0}", level);
  28.         }
  30.         // после загрузки игрового уровня
  31.         public override void OnLevelLoaded(string level) {
  32.             BusEngine.Log.Info("MyPlugin OnLevelLoaded: {0}", level);
  33.         }
  35.         // когда игрок может управлять главным героем - время игры идёт
  36.         public override void OnGameStart() {
  37.             BusEngine.Log.Info("MyPlugin OnGameStart");
  38.         }
  40.         // когда время остановлено - пауза или закрытие уровня
  41.         public override void OnGameStop() {
  42.             BusEngine.Log.Info("MyPlugin OnGameStop");
  43.         }
  45.         // вызывается при отрисовки каждого кадра
  46.         public override void OnGameUpdate() {
  47.             BusEngine.Log.Info("MyPlugin OnGameUpdate");
  48.         }
  50.         // когда игрок начинает подключаться к серверу
  51.         public override void OnClientConnectionReceived(int channelId) {
  52.             BusEngine.Log.Info("MyPlugin OnClientConnectionReceived: {0}", channelId);
  53.         }
  55.         // когда игрок подключился к серверу
  56.         public override void OnClientReadyForGameplay(int channelId) {
  57.             BusEngine.Log.Info("MyPlugin OnClientReadyForGameplay: {0}", channelId);
  58.         }
  60.         // когда игрока выкинуло из сервера - обрыв связи с сервером или он отключился сам
  61.         public override void OnClientDisconnected(int channelId) {
  62.             BusEngine.Log.Info("MyPlugin OnClientDisconnected: {0}", channelId);
  63.         }
  64.     }
  65.     /** API BusEngine.Plugin */
  66. }
  67. /** API BusEngine.Game - пользовательский код */
